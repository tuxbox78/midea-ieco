# Pinned dependency versions for midea-ieco.
#
# msmart-ng and midea-local track Midea's cloud and device firmware and have
# shipped breaking API changes between releases (see "Development lessons
# learned" in the README). They are therefore pinned to the exact versions
# validated for this release rather than floated on every install.
#
# To move up: bump the versions here, run `bash tests/run_all.sh`, verify
# against a real device, then commit.
msmart-ng==2026.7.0
midea-local==6.10.0

# midea-local 6.10.0 imports typing_extensions ("from typing_extensions import
# deprecated" in midealocal/device.py) but does NOT declare it as a dependency,
# so pip never pulls it in and `python -m midealocal.cli` crashes with
# ModuleNotFoundError on current Python (observed on 3.13/3.14). Pin it here
# explicitly as a workaround until midea-local declares it upstream.
typing_extensions==4.16.0
